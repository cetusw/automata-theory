<mainblock> -> main <body> end.
<body> -> <decls>; begin <sts> end
<decls> -> <decl> | <decl> ; <decls>
<decl> -> <var> | <consts>
<var> -> var <id> : <type>
<id> -> id | id , <id>
<type> -> int | float
<consts> -> <const> | <const> ; <consts>
<const> -> id = <expr>
<sts> -> <st> | <st> ; <sts>
<st> -> <assign>
<assign> -> id := <expr>
<expr> -> <exp> + <T> | <T>
<T> -> <T> * <T> | <F>
<F> -> -<F> | (<expr>) | id | <number>
